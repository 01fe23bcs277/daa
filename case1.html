// Rabin-Karp Algorithm to find a pattern in a text
void rabinKarp(string text, string pattern) {
    int M = pattern.length();
    int N = text.length();
    int d = 256; // Number of characters in the input alphabet
    int q = 101; // A prime number
    int h = 1;
    int p = 0; // hash value for pattern
    int t = 0; // hash value for text

    // The value of h would be "pow(d, M-1) % q"
    for (int i = 0; i < M - 1; i++)
        h = (h * d) % q;

    // Calculate the hash value for the pattern and the first window of text
    for (int i = 0; i < M; i++) {
        p = (d * p + pattern[i]) % q;
        t = (d * t + text[i]) % q;
    }

    // Slide the pattern over text one by one
    for (int s = 0; s <= N - M; s++) {
        // Check if hash values are the same
        if (p == t) {
            int j;
            // If hash values match, check character by character
            for (j = 0; j < M; j++) {
                if (text[s + j] != pattern[j])
                    break;
            }
            if (j == M)
                cout << "Pattern found at index " << s << endl;
        }

        // Calculate hash value for next window of text
        if (s < N - M) {
            t = (d * (t - text[s] * h) + text[s + M]) % q;
            if (t < 0)
                t = (t + q);
        }
    }
}

int main() {
    string text = "GEEKS FOR GEEKS";
    string pattern = "GEEKS";
    rabinKarp(text, pattern);
    return 0;
}

// KMP Algorithm to find the pattern in a string
void KMP(string text, string pattern) {
    int M = pattern.length();
    int N = text.length();
    vector lps(M, 0); // Longest Prefix Suffix array

    // Preprocess the pattern (calculate lps array)
    int j = 0;
    for (int i = 1; i < M; i++) {
        if (pattern[i] == pattern[j]) {
            j++;
            lps[i] = j;
        } else {
            if (j != 0) {
                j = lps[j - 1];
                i--;
            } else {
                lps[i] = 0;
            }
        }
    }

    // Search the pattern in the text using the lps array
    int i = 0;
    j = 0;
    while (i < N) {
        if (pattern[j] == text[i]) {
            i++;
            j++;
        }
        if (j == M) {
            cout << "Pattern found at index " << i - j << endl;
            j = lps[j - 1];
        }
        else if (i < N && pattern[j] != text[i]) {
            if (j != 0) {
                j = lps[j - 1];
            } else {
                i++;
            }
        }
    }
}

int main() {
    string text = "ABABDABACDABABCABAB";
    string pattern = "ABABCABAB";
    KMP(text, pattern);
    return 0;
}
